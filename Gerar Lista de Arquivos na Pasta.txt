Get-ChildItem *.pdf | Sort-Object Name | ForEach-Object {
  $pdf = $_.Name
  $docx = [System.IO.Path]::ChangeExtension($pdf, ".docx")

  if (Test-Path $docx) {
    "<li><a href=""$pdf"" target=""_blank"">$pdf</a> <span class=""meta""> | <a href=""$docx"" target=""_blank"">Resumo (DOCX)</a></span></li>"
  } else {
    "<li><a href=""$pdf"" target=""_blank"">$pdf</a></li>"
  }
} | Out-File itens.html -Encoding utf8

Write-Host "OK: itens.html gerado."
